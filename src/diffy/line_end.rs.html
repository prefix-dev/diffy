3#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
4pub enum LineEnd {
5    /// Line Feed (LF) - Common on Unix, Linux, and macOS (`\n`).
6    Lf,
7    /// Carriage Return + Line Feed (CRLF) - Used on Windows (`\r\n`).
8    CrLf,
9}
10
11impl From<LineEnd> for &str {
12    fn from(value: LineEnd) -> Self {
13        match value {
14            LineEnd::Lf => "\n",
15            LineEnd::CrLf => "\r\n",
16        }
17    }
18}
19
20impl From<LineEnd> for &[u8] {
21    fn from(value: LineEnd) -> Self {
22        match value {
23            LineEnd::Lf => b"\n",
24            LineEnd::CrLf => b"\r\n",
25        }
26    }
27}

29impl LineEnd {
30    /// Strip only line ending from `line`. Returns pair of stripped line and optionally stripped line ending.
31    ///
32    /// Assumes that if line has line ending, then it is last chars.
33    pub fn strip<T: ?Sized + Text + ToOwned>(line: &T) -> (&T, Option<LineEnd>) {
34        let mut line_ending = None;
35        let line_without_lf = line.strip_suffix("\n").inspect(|_| {
36            line_ending = Some(LineEnd::Lf);
37        });
38        let line_without_crlf = line_without_lf
39            .and_then(|line| line.strip_suffix("\r"))
40            .inspect(|_| {
41                line_ending = Some(LineEnd::CrLf);
42            });
43        let stripped_line = line_without_crlf.or(line_without_lf);
44
45        (stripped_line.unwrap_or(line), line_ending)
46    }
47
48    /// Choose line ending based on the scores.
49    pub fn choose_from_scores(lf_score: usize, crlf_score: usize) -> LineEnd {
50        #[allow(clippy::if_same_then_else)]
51        if lf_score > crlf_score {
52            LineEnd::Lf
53        } else if lf_score < crlf_score {
54            LineEnd::CrLf
55        } else if cfg!(windows) {
56            LineEnd::CrLf
57        } else {
58            LineEnd::Lf
59        }
60    }
61
62    /// Returns most common line ending.
63    pub fn most_common<T: ?Sized + Text + ToOwned>(input: &T) -> LineEnd {
64        let mut lf_score: usize = 0;
65        let mut crlf_score: usize = 0;
66
67        let mut previous_is_cr = false;
68        for byte in input.as_bytes() {
69            match byte {
70                b'\r' => {
71                    previous_is_cr = true;
72                }
73                b'\n' => {
74                    if previous_is_cr {
75                        crlf_score += 1;
76                    } else {
77                        lf_score += 1;
78                    }
79                    previous_is_cr = false;
80                }
81                _ => {
82                    previous_is_cr = false;
83                    continue;
84                }
85            }
86        }
87
88        LineEnd::choose_from_scores(lf_score, crlf_score)
89    }
90}

97    #[rstest]
98    #[case("hello")]
99    #[case("\r")]
100    #[case("")]
101    #[case("\rhello")]
102    #[case("hello \r")]
103    #[case("\r\nhello")]

106    #[case("hello\r\n ")]
107    fn strip_no_line_ending(#[case] input: &str) {
108        let stripped = LineEnd::strip(input);
109        assert_eq!((input, None), stripped);
110    }
111
112    #[rstest]
113    #[case("hello\n")]
114    #[case("hello\r\n")]
115    #[case("hello \n")]
116    #[case("hello \r\n")]
117    #[case("\r\nhello \n")]
118    #[case("hello \r\n")]
119    fn strip_line_ending(#[case] input: &str) {
120        let (stripped, line_ending) = LineEnd::strip(input);
121        assert!(
122            input.len().saturating_sub(2) <= stripped.len()
123                && stripped.len() < input.len()
124                && line_ending.is_some(),
125            "Expected no newline at the end, but got: {:#?}\nOriginal line is: {:#?}",
126            stripped,
127            input
128        );
129    }
