13/// Kind of line start in `Hunk` header.
14#[derive(Debug)]
15pub enum HeaderLineKind {
16    Adding,
17    Removing,
18}
19
20impl fmt::Display for HeaderLineKind {
21    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
22        write!(
23            f,
24            "{}",
25            match self {
26                HeaderLineKind::Adding => "+++",
27                HeaderLineKind::Removing => "---",
28            }
29        )
30    }
31}

33/// An error returned when parsing a `Patch` using [`Patch::from_str`] fails
34///
35/// [`Patch::from_str`]: struct.Patch.html#method.from_str
36#[derive(thiserror::Error, Debug)]
37pub enum ParsePatchError {
38    #[error("unexpected end of file")]
39    UnexpectedEof,
40    #[error("multiple '{0}' lines")]
41    HeaderMultipleLines(HeaderLineKind),
42    #[error("unable to parse filename")]
43    UnableToParseFilename,
44    #[error("filename unterminated")]
45    UnterminatedFilename,
46    #[error("invalid char in unquoted filename")]
47    InvalidCharInUnquotedFilename,
48    #[error("expected escaped character")]
49    ExpectedEscapedCharacter,
50    #[error("invalid escaped character")]
51    InvalidEscapedCharacter,
52    #[error("invalid unescaped character")]
53    InvalidUnescapedCharacter,
54    #[error("no hunks found")]
55    NoHunks,
56    #[error("hunks not in order or overlap")]
57    HunksOrder,
58    #[error("hunk header does not match hunk")]
59    HunkHeaderHunkMismatch,
60    #[error("unable to parse hunk header")]
61    HunkHeader,
62    #[error("hunk header unterminated")]
63    HunkHeaderUnterminated,
64    #[error("can't parse range")]
65    Range,
66    #[error("expected end of hunk")]
67    ExpectedEndOfHunk,
68    #[error("expected no more deleted lines")]
69    UnexpectedDeletedLine,
70    #[error("expected no more inserted lines")]
71    UnexpectedInsertLine,
72    #[error("unexpected 'No newline at end of file' line")]
73    UnexpectedNoNewlineAtEOF,
74    #[error("unexpected line in hunk body")]
75    UnexpectedLineInHunkBody,
76    #[error("missing newline")]
77    MissingNewline,
78}

80#[derive(Debug, Clone, Default)]
81pub enum HunkRangeStrategy {
82    /// Do not trust the line counts in the hunk headers and check
83    /// that they are indeed match hunk lines.
84    #[default]
85    Check,
86    /// Do not trust the line counts in the hunk headers, but infer
87    /// them by inspecting the patch (e.g. after editing the patch
88    /// without adjusting the hunk headers appropriately).
89    ///
90    /// Note that we also skip all empty lines at then end of the hunk
91    /// lines before calculating hunk ranges.
92    ///
93    /// Tries to resemble behavior of `git apply` `--recount`
94    /// argument.
95    Recount,
96    /// Trust the line counts in the hunk headers.
97    Ignore,
98}

106struct Parser<'a, T: Text + ?Sized> {
107    lines: std::iter::Peekable<LineIter<'a, T>>,
108    config: ParserConfig,
109}

111impl<'a, T: Text + ?Sized> Parser<'a, T> {
112    fn new(input: &'a T) -> Self {
113        Self::with_config(input, ParserConfig::default())
114    }
115
116    fn with_config(input: &'a T, config: ParserConfig) -> Self {
117        Self {
118            lines: LineIter::new(input).peekable(),
119            config,
120        }
121    }
122
123    fn peek(&mut self) -> Option<&(&'a T, Option<LineEnd>)> {
124        self.lines.peek()
125    }
126
127    fn next(&mut self) -> Result<(&'a T, Option<LineEnd>)> {
128        let line = self.lines.next().ok_or(ParsePatchError::UnexpectedEof)?;
129        Ok(line)
130    }
131}

137pub fn parse_multiple_with_config(input: &str, config: ParserConfig) -> Result<Vec<Diff<'_, str>>> {
138    let mut parser = Parser::with_config(input, config);
139    let mut patches = vec![];
140    loop {
141        match (patch_header(&mut parser), hunks(&mut parser)) {
142            (Ok(header), Ok(hunks)) => {
143                let original = header.0.map(|(line, _end)| convert_cow_to_str(line));
144                let modified = header.1.map(|(line, _end)| convert_cow_to_str(line));
145                patches.push(Diff::new(original, modified, hunks))
146            }
147            (Ok((None, None)), Err(_)) => break,
148            // Allow NoHunks error when we have valid headers (pure renames/deletes/adds)
149            (Ok(header), Err(ParsePatchError::NoHunks))
150                if header.0.is_some() || header.1.is_some() =>
151            {
152                let original = header.0.map(|(line, _end)| convert_cow_to_str(line));
153                let modified = header.1.map(|(line, _end)| convert_cow_to_str(line));
154                patches.push(Diff::new(original, modified, vec![]))
155            }
156            (Ok(_), Err(e)) | (Err(e), _) => {
157                return Err(e);
158            }
159        }
160    }
161    Ok(patches)
162}

222fn convert_cow_to_str(cow: Cow<'_, [u8]>) -> Cow<'_, str> {
223    match cow {
224        Cow::Borrowed(b) => std::str::from_utf8(b).unwrap().into(),

226    }

229#[allow(clippy::type_complexity)]
230fn patch_header<'a, T: Text + ToOwned + ?Sized>(
231    parser: &mut Parser<'a, T>,
232) -> Result<(
233    Option<(Cow<'a, [u8]>, Option<LineEnd>)>,
234    Option<(Cow<'a, [u8]>, Option<LineEnd>)>,
235)> {
236    let (git_original, git_modified, saw_git_header) = header_preamble(parser)?;
237
238    let mut filename1 = None;
239    let mut filename2 = None;
240    let mut saw_traditional_header1 = false;
241    let mut saw_traditional_header2 = false;
242
243    while let Some((line, _end)) = parser.peek() {
244        if line.starts_with("--- ") {
245            if saw_traditional_header1 {
246                return Err(ParsePatchError::HeaderMultipleLines(
247                    HeaderLineKind::Removing,
248                ));
249            }
250            saw_traditional_header1 = true;
251            filename1 = parse_filename("--- ", parser.next()?, saw_git_header)?;
252        } else if line.starts_with("+++ ") {
253            if saw_traditional_header2 {
254                return Err(ParsePatchError::HeaderMultipleLines(HeaderLineKind::Adding));
255            }
256            saw_traditional_header2 = true;
257            filename2 = parse_filename("+++ ", parser.next()?, saw_git_header)?;
258        } else {
259            break;
260        }
261    }
262
263    // Traditional --- +++ headers take precedence over git metadata
264    // If we saw a traditional header (even if it parsed to None for /dev/null), use it
265    // Otherwise fall back to git metadata
266    let original = if saw_traditional_header1 {
267        filename1
268    } else {
269        git_original
270    };
271    let modified = if saw_traditional_header2 {
272        filename2
273    } else {
274        git_modified
275    };
276
277    Ok((original, modified))
278}

280// Parse the patch header preamble, extracting filenames from git metadata.
281// Skips preamble lines like "diff --git", git metadata, etc., until reaching
282// the first filename header ("--- " or "+++ ") or hunk line.
283// Returns extracted filenames from git metadata (for pure renames/deletes/adds)
284// and a flag indicating whether we saw a git header (for prefix stripping).
285#[allow(clippy::type_complexity)]
286fn header_preamble<'a, T: Text + ToOwned + ?Sized>(
287    parser: &mut Parser<'a, T>,
288) -> Result<(
289    Option<(Cow<'a, [u8]>, Option<LineEnd>)>,
290    Option<(Cow<'a, [u8]>, Option<LineEnd>)>,
291    bool, // saw_git_header
292)> {
293    let mut git_original = None;
294    let mut git_modified = None;
295    let mut rename_from = None;
296    let mut rename_to = None;
297    let mut seen_diff_git = false;
298
299    while let Some((line, end)) = parser.peek() {
300        if line.starts_with("--- ") | line.starts_with("+++ ") | line.starts_with("@@ ") {
301            break;
302        }
303
304        // Parse git diff header: diff --git a/... b/...
305        if line.starts_with("diff --git ") {
306            // If we've already seen a diff --git line and have extracted data,
307            // this is the start of the next patch, so stop here
308            if seen_diff_git && (git_original.is_some() || rename_from.is_some()) {
309                break;
310            }
311            seen_diff_git = true;
312
313            if let Some(rest) = line.strip_prefix("diff --git ") {
314                // Parse "a/file1 b/file2" (with prefixes) or "file1 file2" (without prefixes)
315                // Try to split on " b/" first to detect standard format
316                if let Some((file1, file2)) = rest.split_at_exclusive(" b/") {
317                    // Standard format with b/ prefix
318                    git_original = parse_git_filename(file1, true).map(|f| (f, *end));
319                    git_modified = parse_git_filename(file2, true).map(|f| (f, *end));
320                } else if let Some((file1, file2)) = rest.split_at_exclusive(" ") {
321                    // --no-prefix format
322                    git_original = parse_git_filename(file1, false).map(|f| (f, *end));
323                    git_modified = parse_git_filename(file2, false).map(|f| (f, *end));
324                }
325                // If neither split works, skip this line (malformed diff --git line)
326            }
327        }
328        // Parse rename from/to
329        else if line.starts_with("rename from ")
330            && let Some(filename) = line.strip_prefix("rename from ")
331        {
332            rename_from = Some((Cow::Borrowed(filename.as_bytes()), *end));
333        } else if line.starts_with("rename to ")
334            && let Some(filename) = line.strip_prefix("rename to ")
335        {
336            rename_to = Some((Cow::Borrowed(filename.as_bytes()), *end));
337        }
338
339        parser.next()?;
340    }
341
342    // Prefer rename from/to over git diff header
343    let original = rename_from.or(git_original);
344    let modified = rename_to.or(git_modified);
345
346    Ok((original, modified, seen_diff_git))
347}

349#[allow(clippy::type_complexity)]
350fn parse_filename<'a, T: Text + ToOwned + ?Sized>(
351    prefix: &str,
352    l: (&'a T, Option<LineEnd>),
353    saw_git_header: bool,
354) -> Result<Option<(Cow<'a, [u8]>, Option<LineEnd>)>> {
355    let line =
356        l.0.strip_prefix(prefix)
357            .ok_or(ParsePatchError::UnableToParseFilename)?;
358
359    let filename = if let Some((filename, _)) = line.split_at_exclusive("\t") {
360        filename
361    } else if let Some((filename, _)) = line.split_at_exclusive(" ") {
362        filename
363    } else if let Some((filename, _)) = line.split_at_exclusive("\n") {
364        filename
365    } else {
366        line
367    };
368
369    // Check for /dev/null before parsing
370    if filename.as_bytes() == b"/dev/null" {
371        return Ok(None);
372    }
373
374    let mut parsed_filename = if let Some(quoted) = is_quoted(filename) {
375        escaped_filename(quoted)?
376    } else {
377        unescaped_filename(filename)?
378    };
379
380    // Strip a/ or b/ prefix only if we saw a git header (git format uses these prefixes)
381    if saw_git_header && let Cow::Borrowed(bytes) = parsed_filename {
382        if let Some(rest) = std::str::from_utf8(bytes)
383            .ok()
384            .and_then(|s| s.strip_prefix("a/"))
385        {
386            parsed_filename = Cow::Borrowed(rest.as_bytes());
387        } else if let Some(rest) = std::str::from_utf8(bytes)
388            .ok()
389            .and_then(|s| s.strip_prefix("b/"))
390        {
391            parsed_filename = Cow::Borrowed(rest.as_bytes());
392        }
393    }
394
395    Ok(Some((parsed_filename, l.1)))
396}

398fn is_quoted<T: Text + ?Sized>(s: &T) -> Option<&T> {
399    s.strip_prefix("\"").and_then(|s| s.strip_suffix("\""))
400}

402fn unescaped_filename<T: Text + ToOwned + ?Sized>(filename: &T) -> Result<Cow<'_, [u8]>> {
403    // NOTE: may be a problem for other types of line feed except "\n" and "\r\n".
404    let bytes = filename.as_bytes().trim_ascii_end();
405
406    if bytes.iter().any(|b| ESCAPED_CHARS_BYTES.contains(b)) {
407        return Err(ParsePatchError::InvalidCharInUnquotedFilename);
408    }
409
410    Ok(bytes.into())
411}

413fn escaped_filename<T: Text + ToOwned + ?Sized>(escaped: &T) -> Result<Cow<'_, [u8]>> {
414    let mut filename = Vec::new();
415
416    let mut chars = escaped.as_bytes().iter().copied();
417    while let Some(c) = chars.next() {
418        if c == b'\\' {
419            let ch = match chars
420                .next()
421                .ok_or(ParsePatchError::ExpectedEscapedCharacter)?
422            {
423                b'n' => b'\n',
424                b't' => b'\t',
425                b'0' => b'\0',
426                b'r' => b'\r',
427                b'\"' => b'\"',
428                b'\\' => b'\\',
429                _ => return Err(ParsePatchError::InvalidEscapedCharacter),
430            };
431            filename.push(ch);
432        } else if ESCAPED_CHARS_BYTES.contains(&c) {
433            return Err(ParsePatchError::InvalidUnescapedCharacter);
434        } else {
435            filename.push(c);
436        }
437    }
438
439    Ok(filename.into())
440}

442/// Parse a filename from a git diff header, handling:
443/// - Standard format with a/ and b/ prefixes
444/// - --no-prefix format without prefixes
445/// - /dev/null for created/deleted files
446///
447/// Returns None for /dev/null (represents non-existent file)
448fn parse_git_filename<T: Text + ?Sized>(filename: &T, has_prefix: bool) -> Option<Cow<'_, [u8]>> {
449    // Check for /dev/null (file doesn't exist)
450    if filename.as_bytes() == b"/dev/null" {
451        return None;
452    }
453
454    // If we detected prefixes (found " b/" in the line), strip a/ or b/
455    if has_prefix {
456        if let Some(stripped) = filename.strip_prefix("a/") {
457            return Some(Cow::Borrowed(stripped.as_bytes()));
458        } else if let Some(stripped) = filename.strip_prefix("b/") {
459            return Some(Cow::Borrowed(stripped.as_bytes()));
460        }
461    }
462
463    // No prefix or couldn't strip, use as-is
464    Some(Cow::Borrowed(filename.as_bytes()))
465}

467fn verify_hunks_in_order<T: ?Sized + ToOwned>(hunks: &[Hunk<'_, T>]) -> bool {
468    for hunk in hunks.windows(2) {
469        if hunk[0].old_range.end() > hunk[1].old_range.start()
470            || hunk[0].new_range.end() > hunk[1].new_range.start()
471        {
472            return false;
473        }
474    }
475    true
476}

478fn hunks<'a, T: Text + ?Sized + ToOwned>(parser: &mut Parser<'a, T>) -> Result<Vec<Hunk<'a, T>>> {
479    let mut hunks = Vec::new();
480    while parser.peek().is_some() {
481        let r = hunk(parser);
482
483        // TODO: Handle properly. For example there is case where hunk
484        // is partially parsed. I think we want to make it hard error
485        // instead or treating it as PS.
486        if let Ok(h) = r {
487            hunks.push(h);
488        } else {
489            break;
490        }
491    }
492
493    if hunks.is_empty() {
494        return Err(ParsePatchError::NoHunks);
495    }
496
497    // check and verify that the Hunks are in sorted order and don't overlap
498    if !verify_hunks_in_order(&hunks) {
499        return Err(ParsePatchError::HunksOrder);
500    }
501
502    Ok(hunks)
503}

505// Hunk ranges tolerance levels based on the end lines.
506fn tolerance_level<T: Text + ?Sized + ToOwned>(lines: &[Line<'_, T>]) -> (usize, bool) {
507    let mut tolerance = 0;
508    let mut revlines = lines.iter().rev();
509    while let Some(Line::Context((_, end))) = revlines.next() {
510        if end.is_some() {
511            tolerance += 1;
512        } else {
513            break;
514        }
515    }
516
517    let line_ends_with_newline = matches!(revlines.next(), Some(Line::Context((_, Some(_)))));
518
519    (tolerance, line_ends_with_newline)
520}

522fn hunk<'a, T: Text + ?Sized + ToOwned>(parser: &mut Parser<'a, T>) -> Result<Hunk<'a, T>> {
523    let n = *parser.peek().ok_or(ParsePatchError::UnexpectedEof)?;
524    let (mut range1, mut range2, function_context) = hunk_header(n)?;
525    let _ = parser.next();
526    let mut lines = hunk_lines(parser, &range1, &range2)?;
527
528    // check counts of lines to see if they match the ranges in the hunk header
529    let (len1, len2) = super::hunk_lines_count(&lines);
530
531    match parser.config.hunk_strategy {
532        HunkRangeStrategy::Check => {
533            let t = tolerance_level(&lines);
534            let tolerance = t.0 + usize::from(t.1);
535
536            if len1.abs_diff(range1.len) > tolerance || len2.abs_diff(range2.len) > tolerance {
537                return Err(ParsePatchError::HunkHeaderHunkMismatch);
538            }
539        }
540        HunkRangeStrategy::Recount => {
541            let empty_context_lines = lines
542                .iter()
543                .rev()
544                .take_while(|l| match *l {
545                    Line::Context(c) => c.0.len() == 0 && c.1.is_some(),
546                    _ => false,
547                })
548                .count();
549
550            lines = lines
551                .into_iter()
552                .rev()
553                .skip(empty_context_lines)
554                .rev()
555                .collect();
556
557            // Should never overflow since len{1,2} >= empty_context_lines by the defintion above.
558            range1.len = len1 - empty_context_lines;
559            range2.len = len2 - empty_context_lines;
560        }
561        HunkRangeStrategy::Ignore => (),
562    }
563
564    Ok(Hunk::new(range1, range2, function_context, lines))
565}

567type HunkHeader<'a, T> = (HunkRange, HunkRange, Option<(&'a T, Option<LineEnd>)>);
568
569fn hunk_header<T: Text + ?Sized>(oinput: (&T, Option<LineEnd>)) -> Result<HunkHeader<'_, T>> {
570    let input = oinput
571        .0
572        .strip_prefix("@@ ")
573        .ok_or(ParsePatchError::HunkHeader)?;
574
575    let (ranges, function_context) = input
576        .split_at_exclusive(" @@")
577        .ok_or(ParsePatchError::HunkHeaderUnterminated)?;
578    let function_context = function_context.strip_prefix(" ");
579
580    let (range1, range2) = ranges
581        .split_at_exclusive(" ")
582        .ok_or(ParsePatchError::HunkHeader)?;
583    let range1 = range(
584        range1
585            .strip_prefix("-")
586            .ok_or(ParsePatchError::HunkHeader)?,
587    )?;
588    let range2 = range(
589        range2
590            .strip_prefix("+")
591            .ok_or(ParsePatchError::HunkHeader)?,
592    )?;
593    Ok((range1, range2, function_context.map(|fc| (fc, oinput.1))))
594}
595
596fn range<T: Text + ?Sized>(s: &T) -> Result<HunkRange> {
597    let (start, len) = if let Some((start, len)) = s.split_at_exclusive(",") {
598        (
599            start.parse().ok_or(ParsePatchError::Range)?,
600            len.parse().ok_or(ParsePatchError::Range)?,
601        )
602    } else {
603        (s.parse().ok_or(ParsePatchError::Range)?, 1)
604    };
605
606    Ok(HunkRange::new(start, len))
607}

609fn hunk_lines<'a, T: Text + ?Sized + ToOwned>(
610    parser: &mut Parser<'a, T>,
611    old_range: &HunkRange,
612    new_range: &HunkRange,
613) -> Result<Vec<Line<'a, T>>> {
614    let mut lines: Vec<Line<'a, T>> = Vec::new();
615    let mut no_newline_context = false;
616    let mut no_newline_delete = false;
617    let mut no_newline_insert = false;
618
619    // Track how many lines we've seen for each side
620    let mut old_lines_seen = 0;
621    let mut new_lines_seen = 0;
622
623    // Calculate maximum lines we should read based on ranges
624    let expected_old_lines = old_range.len;
625    let expected_new_lines = new_range.len;
626
627    while let Some(line) = parser.peek() {
628        // Check if we've read enough lines based on the ranges,
629        // but continue to check for the "No newline at end of file" marker
630        if old_lines_seen >= expected_old_lines && new_lines_seen >= expected_new_lines {
631            // Check if the next line is the "No newline at end of file" marker
632            if !line.0.starts_with(NO_NEWLINE_AT_EOF) {
633                // We've read all the lines we expect for this hunk
634                break;
635            }
636        }
637
638        let line = if no_newline_context {
639            return Err(ParsePatchError::ExpectedEndOfHunk);
640        } else if let Some(l) = line.0.strip_prefix(" ") {
641            old_lines_seen += 1;
642            new_lines_seen += 1;
643            Line::Context((l, line.1))
644        } else if line.0.len() == 0 && line.1.is_some() {
645            old_lines_seen += 1;
646            new_lines_seen += 1;
647            Line::Context(*line)
648        } else if let Some(l) = line.0.strip_prefix("-") {
649            if no_newline_delete {
650                return Err(ParsePatchError::UnexpectedDeletedLine);
651            }
652            old_lines_seen += 1;
653            Line::Delete((l, line.1))
654        } else if let Some(l) = line.0.strip_prefix("+") {
655            if no_newline_insert {
656                return Err(ParsePatchError::UnexpectedInsertLine);
657            }
658            new_lines_seen += 1;
659            Line::Insert((l, line.1))
660        } else if line.0.starts_with(NO_NEWLINE_AT_EOF) {
661            let last_line = lines
662                .pop()
663                .ok_or(ParsePatchError::UnexpectedNoNewlineAtEOF)?;
664            match last_line {
665                Line::Context((line, _end)) => {
666                    no_newline_context = true;
667                    Line::Context((line, None))
668                }
669                Line::Delete((line, _end)) => {
670                    no_newline_delete = true;
671                    Line::Delete((line, None))
672                }
673                Line::Insert((line, _end)) => {
674                    no_newline_insert = true;
675                    Line::Insert((line, None))
676                }
677            }
678        } else {
679            return Err(ParsePatchError::UnexpectedLineInHunkBody);
680        };
681
682        lines.push(line);
683        parser.next()?;
684    }
685
686    Ok(lines)
687}
